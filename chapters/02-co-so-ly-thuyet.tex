\section{2. Cơ sở lý thuyết}

\begin{frame}[t]{2.1. HPC job scheduling và Slurm request}
\small
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Người dùng khai báo}
\begin{itemize}
  \item partition, account;
  \item nodes, CPU, GPU, memory;
  \item walltime và dependency;
  \item script thực thi.
\end{itemize}
\column{0.48\textwidth}
\textbf{Scheduler quyết định}
\begin{itemize}
  \item admission và priority;
  \item queueing và backfilling;
  \item allocation;
  \item thời điểm thực thi.
\end{itemize}
\end{columns}
\begin{block}{Vị trí của nghiên cứu}
Tinh chỉnh request trước submission; không can thiệp thuật toán scheduling nội bộ.
\end{block}
\end{frame}

\begin{frame}[t]{2.2. Intent, constraints, preferences và flexibility}
\small
\begin{description}
  \item[Goal] Kết quả người dùng muốn đạt.
  \item[Hard constraints] Điều kiện bắt buộc; vi phạm là loại candidate.
  \item[Soft preferences] Tiêu chí chỉ dùng để xếp hạng candidate đã khả thi.
  \item[Authorized flexibility] Trường nào được đổi, miền giá trị và điều kiện đổi.
\end{description}
\[
r\in\mathcal{R}^{\mathrm{feasible}}_j
\quad\Longleftrightarrow\quad
\text{mọi hard constraint tới hạn đều pass}
\]
\begin{alertblock}{Nguyên tắc}
Không dùng preference để bù cho một hard constraint bị vi phạm.
\end{alertblock}
\end{frame}

\begin{frame}[t]{2.3. Moldable resource profiles và candidate space}
\small
\[
\mathcal{R}_j=
\mathcal{C}_{\mathrm{cluster}}
\cap
\mathcal{F}_{\mathrm{authorized}}
\cap
\mathcal{C}_{\mathrm{application}}
\]
\begin{itemize}
  \item $\mathcal{C}_{\mathrm{cluster}}$: tài nguyên và cấu hình thực sự tồn tại.
  \item $\mathcal{F}_{\mathrm{authorized}}$: chỉ các thay đổi người dùng cho phép.
  \item $\mathcal{C}_{\mathrm{application}}$: profile mà ứng dụng có thể chạy đúng.
\end{itemize}
\begin{block}{Moldable, không phải malleable trong phạm vi này}
Chọn một resource profile trước khi nộp; không thay đổi allocation khi job đang chạy.
\end{block}
\end{frame}

\begin{frame}[t]{2.4. Feasibility, evidence và uncertainty}
\small
\[
F(r)=F_{\mathrm{app}}(r)\land F_{\mathrm{system}}(r)\land
F_{\mathrm{policy}}(r)\land F_{\mathrm{quota}}(r)\land F_{\mathrm{intent}}(r)
\]
\[
\mathbb{L}_3=\{\mathrm{pass},\mathrm{fail},\mathrm{unknown}\}
\]
\begin{itemize}
  \item \textbf{pass}: evidence đủ và điều kiện được thỏa.
  \item \textbf{fail}: evidence cho thấy điều kiện bị vi phạm.
  \item \textbf{unknown}: thiếu, cũ, xung đột hoặc độ tin cậy chưa đủ.
\end{itemize}
\begin{alertblock}{Quy tắc an toàn}
Unknown trên ràng buộc tới hạn không được chuyển thành khuyến nghị tự tin.
\end{alertblock}
\end{frame}

\begin{frame}[t]{2.5. Kiểm tra deadline dưới bất định}
\small
\textbf{Điều kiện theo upper bound:}
\[
t+\widehat{T}_{\mathrm{wait}}^{U}(r)+\widehat{T}_{\mathrm{run}}^{U}(r)\le D_j
\]
\textbf{Hoặc điều kiện xác suất:}
\[
\Pr\!\left[T_{\mathrm{wait}}(r)+T_{\mathrm{run}}(r)\le D_j-t\right]\ge\alpha_j
\]
\begin{block}{Diễn giải đúng}
Dự đoán hỗ trợ quyết định với interval, confidence hoặc safety margin; không phải lời bảo đảm hoàn thành.
\end{block}
\end{frame}

\begin{frame}[t]{2.6. Pareto, lexicographic selection và trace}
\small
\begin{center}
\textbf{Hard constraints} $\rightarrow$ \textbf{Evidence gate} $\rightarrow$
\textbf{Pareto} $\rightarrow$ \textbf{Lexicographic} $\rightarrow$ \textbf{Action}
\end{center}
\begin{itemize}
  \item Pareto filtering loại candidate bị trội trên các tiêu chí liên quan.
  \item Lexicographic selection tuân theo thứ tự preference do người dùng khai báo.
  \item Weighted utility chỉ là baseline hoặc evaluation metric, không phải luật chọn chính.
  \item Decision trace lưu input, evidence, validation, dominance/ranking và output.
\end{itemize}
\end{frame}
