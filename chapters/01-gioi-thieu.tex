\section{1. Giới thiệu}

\begin{frame}[t]{1.1. Quy trình nộp công việc HPC}
\small
\begin{center}
\begin{tikzpicture}[node distance=0.55cm, every node/.style={font=\small}]
\tikzstyle{stage}=[draw=BKTemplate, rounded corners, fill=blue!7, minimum height=0.9cm, text width=2.25cm, align=center]
\node[stage] (intent) {Ý định và nhu cầu\\của người dùng};
\node[stage, right=of intent] (script) {Slurm script\\resource request cố định};
\node[stage, right=of script] (sched) {Scheduler\\queue, priority, allocation};
\node[stage, right=of sched] (run) {Phân bổ và\\thực thi};

\draw[->,thick] (script)--(sched);

\end{tikzpicture}
\end{center}
\begin{alertblock}{Vấn đề cốt lõi}
Scheduler nhận request đã cố định, nhưng không trực tiếp nhận mục tiêu, ràng buộc, ưu tiên, mức bất định hay phạm vi linh hoạt mà người dùng chấp thuận.
\end{alertblock}
\end{frame}

\begin{frame}[t]{1.2. Mất mát ngữ nghĩa khi tạo resource request}
\small
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Ý định ban đầu}
\begin{itemize}
  \item mục tiêu: hoàn thành trước deadline;
  \item ràng buộc: GPU, bộ nhớ, quota;
  \item ưu tiên: chi phí trước tốc độ;
  \item linh hoạt: được đổi số node, không đổi GPU;
  \item bất định: runtime chưa được đo.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Thông tin scheduler thấy}
\texttt{--nodes=2}\quad \texttt{--gpus=4}\\
\texttt{--mem=128G}\quad \texttt{--time=02:00:00}
\end{block}
\begin{alertblock}{Semantic loss}
Quá trình dịch thủ công nén nhiều ý nghĩa thành một cấu hình duy nhất và che khuất các phương án được phép.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}[t]{1.3. Giới hạn của request truyền thống}
\small
\begin{itemize}
  \item Request cho biết \emph{đã yêu cầu gì}, không cho biết \emph{vì sao}.
  \item Hard constraints và soft preferences thường không được phân biệt.
  \item Scheduler không biết tham số nào được phép thay đổi.
  \item Một dự đoán runtime không phải là bảo đảm deadline.
  \item Thiếu evidence và provenance khiến khuyến nghị khó kiểm tra lại.
\end{itemize}
\begin{exampleblock}{Ranh giới trách nhiệm}
Lớp đề xuất hoạt động \textbf{trước khi nộp job}. Scheduler vẫn quản lý hàng đợi, priority, backfilling, allocation và execution.
\end{exampleblock}
\end{frame}

\begin{frame}[t]{1.4. Bài toán nghiên cứu}
\small
\begin{block}{Câu hỏi trung tâm}
Làm thế nào chuyển ý định HPC không đầy đủ thành các resource profile được người dùng cho phép, khả thi theo bằng chứng hiện có, và có thể kiểm tra trước khi nộp job?
\end{block}
\begin{block}{Luận điểm trung tâm}
Nghiên cứu đề xuất một \textbf{khung khuyến nghị resource profile nhận biết ý định ở giai đoạn pre-submission}, dựa trên mô hình ý định có cấu trúc, kiểm tra khả thi có bằng chứng, xử lý bất định và decision trace có thể audit.
\end{block}
\end{frame}

\begin{frame}[t,shrink=8]{1.5. Câu hỏi nghiên cứu và phạm vi}
\small
\begin{enumerate}
  \item[\textbf{RQ1.}] Biểu diễn ý định công việc HPC bằng ngôn ngữ tự nhiên thành mô hình ý định có cấu trúc và có thể kiểm tra như thế nào?
  \item[\textbf{RQ2.}] Sinh các phương án tài nguyên được ủy quyền mà không vi phạm ràng buộc người dùng, chính sách site hay năng lực cluster như thế nào?
  \item[\textbf{RQ3.}] Dùng bằng chứng và bất định để xác định khuyến nghị an toàn, không an toàn, hay cần hỏi thêm như thế nào?
  \item[\textbf{RQ4.}] Tạo decision trace có thể audit trước khi nộp job như thế nào?
\end{enumerate}
\vspace{0.1cm}
\textbf{Ngoài phạm vi:} thay thế scheduler, tối ưu toàn hệ thống, tự động nộp job, và khẳng định hiệu quả thực nghiệm khi chưa có kết quả.
\end{frame}

\begin{frame}[t]{1.6. Đóng góp dự kiến}
\small
\begin{enumerate}
  \item Một structured intent schema phân tách mục tiêu, hard constraints, preferences và authorized flexibility.
  \item Không gian candidate hữu hạn, được chặn bởi cluster, quyền thay đổi và năng lực ứng dụng.
  \item Kiểm tra khả thi nhiều lớp với logic ba giá trị và evidence gate.
  \item Pareto filtering và lexicographic selection thay cho weighted sum làm quy tắc chính.
  \item Evidence-backed decision trace cho recommendation, ask-back, abstention hoặc approved action.
\end{enumerate}
\end{frame}
