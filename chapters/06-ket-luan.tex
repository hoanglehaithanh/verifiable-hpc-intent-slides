\section{Thảo luận và định hướng tiếp theo}

% Slide 44
\begin{frame}[t,shrink=20]{Những gì chuyên đề đã chứng minh}
\scriptsize
\vspace{0.15em}
\textbf{Đã chứng minh trong phạm vi đo}\par
\begin{itemize}
\item intent có thể được biểu diễn ở field level với safety validation;
\item alternative profiles có thể được xây dựng từ authorized flexibility;
\item abstention giúp đạt 100\% feasibility khi commit;
\item uncertainty có thể chuyển thành operational action;
\item queue-aware profile selection tạo utility dưới resource pressure.
\end{itemize}
\vspace{0.15em}
\textbf{Chưa thể khẳng định}\par
\begin{itemize}
\item semantic understanding hoàn hảo;
\item universal automatic recommendation;
\item universal makespan improvement;
\item generalization cho mọi workload và cluster.
\end{itemize}
\end{frame}

% Slide 45
\begin{frame}[t,shrink=20]{Kết luận và hướng nghiên cứu tiếp theo}
\scriptsize
\vspace{0.15em}
\textbf{Kết luận}\par
\begin{quote}
\emph{Một lớp intent-aware, evidence-backed và uncertainty-aware trước Slurm submission có thể cải thiện chất lượng resource request mà không thay đổi scheduler, với điều kiện hệ thống được phép từ chối tự động quyết định khi evidence chưa đủ.}
\end{quote}
\vspace{0.15em}
\textbf{Hướng tiếp theo}\par
\begin{itemize}
\item intent-guided active profiling;
\item richer workload representation;
\item dynamic queue-state prediction;
\item multi-node và heterogeneous workloads;
\item intent-aware scheduler interface;
\item human-in-the-loop evaluation.
\end{itemize}
\[
\text{Better intent representation}
\rightarrow
\text{safer profile decision}
\rightarrow
\text{better scheduling opportunity}
\]
\end{frame}

\begin{frame}[t,shrink=20]{Threats to validity}
\scriptsize
\vspace{0.15em}
\textbf{Construct}\par
\begin{itemize}
\item exact match quá nghiêm;
\item usage cost chưa phải billing thật;
\item turnaround khác makespan.
\end{itemize}
\vspace{0.15em}
\textbf{Internal}\par
\begin{itemize}
\item queue snapshot có thể stale;
\item treatments chạy tuần tự;
\item pressure chưa phản ánh đầy đủ production.
\end{itemize}
\vspace{0.15em}
\textbf{External}\par
\begin{itemize}
\item single-node V100;
\item một workload family;
\item 40 English intents;
\item 6 unseen jobs.
\end{itemize}
\vspace{0.15em}
\textbf{Statistical}\par
\begin{itemize}
\item số campaign nhỏ;
\item screening predictor có phạm vi hẹp.
\end{itemize}
\end{frame}
