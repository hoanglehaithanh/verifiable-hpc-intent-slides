\section{4. Mô hình và giải pháp đề xuất}

\begin{frame}[t]{4.1. Từ request ban đầu đến khuyến nghị có kiểm tra}
\small
\begin{center}
\begin{tikzpicture}[node distance=0.28cm, every node/.style={font=\scriptsize}]
\tikzstyle{pipelinebox}=[draw=BKTemplate, rounded corners, fill=blue!7,
  minimum height=0.72cm, text width=1.72cm, align=center]
\node[pipelinebox] (input) {Slurm script\\+ intent};
\node[pipelinebox, right=of input] (parse) {Parse intent\\thành IR};
\node[pipelinebox, right=of parse] (candidate) {Sinh candidate\\có giới hạn};
\node[pipelinebox, right=of candidate] (verify) {Kiểm tra\\feasibility};

\draw[->,thick] (input)--(parse);
\draw[->,thick] (parse)--(candidate);
\draw[->,thick] (candidate)--(verify);
\draw[->,thick] (verify)--(route);
\end{tikzpicture}
\end{center}
\vspace{0.3cm}
\begin{block}{Thông điệp chính}
Pipeline biến một request cố định thành tập lựa chọn có kiểm tra, rồi khuyến nghị hoặc hỏi lại dựa trên evidence.
\end{block}
\begin{alertblock}{Ranh giới}
Đây là lớp hỗ trợ trước khi submit. Slurm vẫn quản lý queue, priority, allocation và execution.
\end{alertblock}
\end{frame}

\begin{frame}[t]{4.2. Hệ thống nhận gì và trả về gì?}
\small
\begin{columns}[T]
\column{0.47\textwidth}
\begin{block}{Input}
\begin{itemize}
  \item Slurm script gốc
  \item Intent bằng ngôn ngữ tự nhiên
  \item Quota và account của người dùng
  \item Snapshot cluster và queue
\end{itemize}
\end{block}
\column{0.49\textwidth}
\begin{block}{Output}
\begin{itemize}
  \item Intent đã cấu trúc hóa
  \item Tập resource profile khả thi
  \item Evidence và kết quả kiểm tra
  \item Profile khuyến nghị hoặc câu hỏi cần làm rõ
  \item Decision trace để kiểm tra lại
\end{itemize}
\end{block}
\end{columns}
\begin{exampleblock}{Kết quả mong muốn}
Người dùng nhận một request tốt hơn trước khi submit, không nhận một quyết định scheduling toàn hệ thống.
\end{exampleblock}
\end{frame}

Soft preference & Tiết kiệm quota \\

\end{columns}

\end{alertblock}

\begin{frame}[t]{4.5. Vai trò và ranh giới của LLM}
\small
\[
(S_j,U_j)\xrightarrow{\mathrm{LLM}}I_j
\]
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{LLM thực hiện}
\begin{itemize}
  \item trích xuất goal;
  \item nhận diện constraints;
  \item tách hard/soft;
  \item phát hiện flexibility và ambiguity;
  \item sinh IR theo schema.
\end{itemize}
\column{0.48\textwidth}
\textbf{LLM không thực hiện}
\begin{itemize}
  \item xác nhận resource/policy/quota;
  \item chứng minh deadline;
  \item chọn final profile;
  \item submission không phê duyệt.
\end{itemize}
\end{columns}
\end{frame}

Cluster hỗ trợ; application chạy được; người dùng cho phép thay đổi; không vi phạm quota hoặc policy.
\end{block}

\begin{tabular}{lrrrrp{0.29\linewidth}}

\midrule

\end{alertblock}
\end{frame}

\begin{frame}[t]{4.7. Kiểm tra khả thi nhiều lớp}
\small
\[
F(r)=F_{\mathrm{app}}(r)\land F_{\mathrm{system}}(r)\land
F_{\mathrm{policy}}(r)\land F_{\mathrm{quota}}(r)\land F_{\mathrm{intent}}(r)
\]
\begin{center}
Mỗi check trả về $\mathbb{L}_3=\{\mathrm{pass},\mathrm{fail},\mathrm{unknown}\}$.
\end{center}
\begin{itemize}
  \item \textbf{fail}: loại candidate và ghi lý do.
  \item \textbf{unknown, không tới hạn}: giữ cờ bất định để xếp hạng/giải thích.
  \item \textbf{unknown, tới hạn}: ask-back, thu thập evidence hoặc abstain.
\end{itemize}
\end{frame}

\begin{frame}[t]{4.8. Cổng evidence và xử lý bất định}
\small
\[
E=(v,t_{\mathrm{obs}},\mathrm{TTL},\mathrm{source})
\]
\begin{itemize}
  \item provenance và source authority;
  \item freshness: $t_{\mathrm{obs}}$ và TTL / \texttt{valid\_until};
  \item confidence, model/policy version;
  \item critical unknowns và xung đột giữa nguồn;
  \item prediction interval hoặc safety margin.
\end{itemize}
\begin{alertblock}{Điều kiện đi qua cổng evidence}
Candidate chỉ đi tiếp khi evidence cho các điều kiện tới hạn đủ mới, đủ tin cậy và không xung đột chưa giải quyết.
\end{alertblock}
\end{frame}

\begin{frame}[t]{4.9. Lọc và xếp hạng profile}
\small
\[
\mathcal{P}_j=\left\{r\in\mathcal{R}^{\mathrm{feasible}}_j
\mid\nexists r':r'\prec r\right\}
\]
\begin{enumerate}
  \item Loại mọi candidate không feasible hoặc không qua evidence gate.
  \item Pareto filtering giữ các candidate không bị dominated.
  \item Lexicographic selection áp dụng thứ tự preference trong $I_j$.
  \item Nếu preference thiếu hoặc hòa, trình bày lựa chọn và hỏi người dùng.
\end{enumerate}
\begin{block}{Baseline, không phải luật chính}
Weighted-sum utility chỉ dùng để so sánh trong thực nghiệm hoặc sensitivity analysis.
\end{block}
\end{frame}

\begin{frame}[t]{4.10. Bốn hành động đầu ra}
\small
\begin{center}
\begin{tabular}{p{0.34\linewidth}p{0.56\linewidth}}
\toprule
\textbf{Action} & \textbf{Điều kiện điển hình} \\
\midrule
\texttt{recommend(profile)} & Candidate khả thi, evidence đủ, preference phân giải được \\
\texttt{ask\_back(question)} & Thiếu intent hoặc cần người dùng chọn trade-off \\
\texttt{abstain(reason)} & Critical unknown, evidence xung đột hoặc không đủ tin cậy \\
\texttt{keep\_original\_request} & Không có phương án tốt hơn được phép và xác nhận \\
\bottomrule
\end{tabular}
\end{center}
\end{frame}

\begin{frame}[t]{4.11. Vết quyết định gắn với evidence}
\small
\begin{itemize}
  \item original request và parsed intent $I_j$;
  \item candidate set $\mathcal{R}_j$ và cách sinh;
  \item evidence source, freshness, confidence;
  \item validation result $V_j$ và critical unknowns;
  \item Pareto dominance / lexicographic ranking;
  \item final recommendation hoặc lý do ask-back/abstain;
  \item timestamp, model, policy và predictor versions.
\end{itemize}
\begin{block}{Tính kiểm tra được}
Trace ghi các artifact và kết quả kiểm tra có thể audit; không ghi hay yêu cầu chain-of-thought nội bộ.
\end{block}
\end{frame}
